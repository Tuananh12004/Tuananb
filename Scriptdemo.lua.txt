local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()
local Window = Rayfield:CreateWindow({
    Name = "DragonBlox Ultimate ",
    LoadingTitle = "Auto Farm DBU 2026",
    LoadingSubtitle = "Tuan_codon",
    ConfigurationSaving = { Enabled = false },
    KeySystem = false
})

local Tab = Window:CreateTab("Auto Farm", nil)

getgenv().AutoFarm = false
getgenv().AutoRebirth = false
getgenv().AutoTransform = false
getgenv().GodMode = false
getgenv().FarmSpeed = 0.08   -- Nhanh hơn tí
getgenv().FarmRange = 300    -- Tăng default để detect xa

-- ==================== AUTO FARM FIXED (quét nhanh + fallback) ====================
Tab:CreateToggle({
    Name = "Auto Farm Mobs / Quests",
    CurrentValue = false,
    Callback = function(Value)
        getgenv().AutoFarm = Value
        Rayfield:Notify({Title = "Auto Farm", Content = Value and "BẬT - Teleport & kill mobs gần nhất..." or "TẮT", Duration = 5})
        
        if Value then
            spawn(function()
                local Player = game.Players.LocalPlayer
                -- Tăng simulation radius ngay từ đầu để thấy mobs xa
                pcall(function()
                    sethiddenproperty(Player, "SimulationRadius", 10000)
                    sethiddenproperty(Player, "MaxSimulationRadius", 10000)
                end)
                
                while getgenv().AutoFarm do
                    task.wait(getgenv().FarmSpeed)
                    if not Player.Character or not Player.Character:FindFirstChild("HumanoidRootPart") or Player.Character.Humanoid.Health <= 0 then continue end
                    
                    local HRP = Player.Character.HumanoidRootPart
                    local found = false
                    
                    -- Quét nhanh GetChildren() trước (nhanh, ít lag)
                    for _, obj in ipairs(workspace:GetChildren()) do
                        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj:FindFirstChild("HumanoidRootPart") and obj.Humanoid.Health > 0 then
                            if not game.Players:GetPlayerFromCharacter(obj) then
                                local root = obj.HumanoidRootPart
                                local dist = (root.Position - HRP.Position).Magnitude
                                if dist <= getgenv().FarmRange then
                                    found = true
                                    HRP.CFrame = root.CFrame * CFrame.new(0, 6, -6)  -- Tele sát, cao tránh stuck
                                    task.wait(0.1)
                                    
                                    -- Method tốt hơn: Activate tool nếu cầm (melee hitbox)
                                    if Player.Character:FindFirstChildOfClass("Tool") then
                                        Player.Character:FindFirstChildOfClass("Tool"):Activate()
                                    end
                                    
                                    -- Backup damage loop (nếu tool không có)
                                    for i = 1, 4 do
                                        pcall(function() obj.Humanoid:TakeDamage(5000) end)
                                        task.wait(0.04)
                                    end
                                    
                                    print("Farm mob: " .. obj.Name .. " | Dist: " .. math.floor(dist))
                                    break  -- Farm 1 mob gần nhất rồi loop lại (tránh lag)
                                end
                            end
                        end
                    end
                    
                    if not found then
                        Rayfield:Notify({Title = "Debug", Content = "Không thấy mob trong range " .. getgenv().FarmRange .. " studs. Di chuyển gần khu spawn mobs (Earth forest/planet) hoặc tăng range slider!", Duration = 6})
                    end
                end
            end)
        end
    end
})

-- ==================== GOD MODE ====================
Tab:CreateToggle({
    Name = "God Mode (No Hit)",
    CurrentValue = false,
    Callback = function(Value)
        getgenv().GodMode = Value
        if Value then
            spawn(function()
                while getgenv().GodMode do
                    task.wait(0.2)
                    if Player.Character then
                        for _, part in pairs(Player.Character:GetChildren()) do
                            if part.Name:lower():find("hitbox") or part.Name:lower():find("damage") then
                                part:Destroy()
                            end
                        end
                    end
                end
            end)
        end
    end
})

-- ==================== AUTO REBIRTH ====================
Tab:CreateToggle({
    Name = "Auto Rebirth",
    CurrentValue = false,
    Callback = function(Value)
        getgenv().AutoRebirth = Value
        if Value then
            spawn(function()
                local rs = game:GetService("ReplicatedStorage")
                local rebirth = rs:FindFirstChild("Rebirth") or rs:FindFirstChild("reb") or (rs.Package and rs.Package.Events and rs.Package.Events.reb)
                if not rebirth then
                    Rayfield:Notify({Title = "Lỗi Rebirth", Content = "Không tìm thấy remote rebirth (thử Dex Explorer tìm 'rebirth' hoặc 'RebirthEvent' trong ReplicatedStorage).", Duration = 8})
                    return
                end
                
                while getgenv().AutoRebirth do
                    task.wait(5)
                    pcall(function() rebirth:InvokeServer() end)
                    print("Rebirth called")
                end
            end)
        end
    end
})

-- ==================== AUTO TRANSFORM & CHARGE ====================
Tab:CreateToggle({
    Name = "Auto Transform & Charge",
    CurrentValue = false,
    Callback = function(Value)
        getgenv().AutoTransform = Value
        if Value then
            spawn(function()
                local vim = game:GetService("VirtualInputManager")
                while getgenv().AutoTransform do
                    task.wait(1.5)
                    vim:SendKeyEvent(true, "C", false, game) task.wait(1) vim:SendKeyEvent(false, "C", false, game)  -- Charge hold lâu hơn
                    vim:SendKeyEvent(true, "G", false, game) task.wait(0.15) vim:SendKeyEvent(false, "G", false, game)  -- Transform
                end
            end)
        end
    end
})

-- ==================== BUTTON & SLIDER ====================
Tab:CreateButton({
    Name = "Bật Full Auto + God",
    Callback = function()
        getgenv().AutoFarm = true
        getgenv().AutoRebirth = true
        getgenv().AutoTransform = true
        getgenv().GodMode = true
        Rayfield:Notify({Title = "FULL ON", Content = "Farm + God Mode active! Check F9 console.", Duration = 6})
    end
})

Tab:CreateSlider({
    Name = "Farm Range",
    Range = {100, 800},
    Increment = 50,
    CurrentValue = 300,
    Callback = function(v) getgenv().FarmRange = v end
})

Tab:CreateSlider({
    Name = "Farm Delay",
    Range = {0.05, 0.3},
    Increment = 0.05,
    CurrentValue = 0.08,
    Callback = function(v) getgenv().FarmSpeed = v end
})
