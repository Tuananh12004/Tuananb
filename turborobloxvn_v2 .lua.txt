-- DBU Auto Farm GUI FINAL FIX - Tuan_codon (with load catch & wait)

print("Script start - Waiting game load...")
while not game:IsLoaded() do task.wait(1) end
print("Game loaded! Trying to load Rayfield...")

local success, Rayfield = pcall(function()
    return loadstring(game:HttpGet('https://sirius.menu/rayfield'))()
end)

if not success or not Rayfield then
    print("LỖI LOAD RAYFIELD: " .. tostring(Rayfield))  -- In lỗi ra F9
    warn("Rayfield fail - Check executor/HttpGet hoặc dùng script raw thay thế.")
    -- Notify fallback nếu có
    game.StarterGui:SetCore("SendNotification", {
        Title = "Lỗi GUI",
        Text = "Rayfield không load được! Thử executor khác hoặc script raw Sus. Check F9 console.",
        Duration = 10
    })
    return  -- Dừng script nếu GUI fail
end

local Window = Rayfield:CreateWindow({
    Name = "DragonBlox Ultimate ",
    LoadingTitle = "Auto Farm DBU 2026",
    LoadingSubtitle = "Tuan_codon",
    ConfigurationSaving = { Enabled = false },
    KeySystem = false
})

local Tab = Window:CreateTab("Auto Farm", nil)

-- Variables an toàn
getgenv().AutoFarm = false
getgenv().FarmSpeed = 0.08
getgenv().FarmRange = 400

-- Auto Farm toggle (tối ưu từ trước)
Tab:CreateToggle({
    Name = "Auto Farm Mobs / Quests",
    CurrentValue = false,
    Callback = function(Value)
        getgenv().AutoFarm = Value
        Rayfield:Notify({Title = "Auto Farm", Content = Value and "BẬT - Tele + kill (cầm melee tool)!" or "TẮT", Duration = 5})
        
        if Value then
            spawn(function()
                local Player = game.Players.LocalPlayer
                pcall(function()
                    sethiddenproperty(Player, "SimulationRadius", 20000)
                    sethiddenproperty(Player, "MaxSimulationRadius", 20000)
                end)
                
                while getgenv().AutoFarm do
                    task.wait(getgenv().FarmSpeed)
                    local Char = Player.Character
                    if not Char or not Char:FindFirstChild("HumanoidRootPart") or Char.Humanoid.Health <= 0 then continue end
                    
                    local HRP = Char.HumanoidRootPart
                    local found = false
                    
                    for _, obj in ipairs(workspace:GetDescendants()) do
                        if found then break end
                        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj.Humanoid.Health > 0 and not game.Players:GetPlayerFromCharacter(obj) then
                            local root = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso") or obj:FindFirstChild("Head")
                            if root then
                                local dist = (root.Position - HRP.Position).Magnitude
                                if dist <= getgenv().FarmRange then
                                    found = true
                                    HRP.CFrame = root.CFrame * CFrame.new(0, 5, -5)
                                    task.wait(0.1)
                                    
                                    local tool = Char:FindFirstChildOfClass("Tool")
                                    if tool then
                                        for i = 1, 10 do tool:Activate() task.wait(0.05) end
                                    end
                                    
                                    pcall(function()
                                        for i = 1, 5 do obj.Humanoid:TakeDamage(5000) task.wait(0.03) end
                                    end)
                                    
                                    print("Farm: " .. obj.Name .. " | Dist: " .. math.floor(dist))
                                end
                            end
                        end
                    end
                end
            end)
        end
    end
})
