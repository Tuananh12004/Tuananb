-- ==================== AUTO FARM FIXED (damage thật + tool spam) ====================
Tab:CreateToggle({
    Name = "Auto Farm Mobs / Quests",
    CurrentValue = false,
    Callback = function(Value)
        getgenv().AutoFarm = Value
        Rayfield:Notify({Title = "Auto Farm", Content = Value and "BẬT - Tele + kill thật (cầm melee tool)!" or "TẮT", Duration = 5})
        
        if Value then
            spawn(function()
                local Player = game.Players.LocalPlayer
                pcall(function()
                    sethiddenproperty(Player, "SimulationRadius", 20000)
                    sethiddenproperty(Player, "MaxSimulationRadius", 20000)
                end)
                
                while getgenv().AutoFarm do
                    task.wait(getgenv().FarmSpeed)
                    if not Player.Character or not Player.Character:FindFirstChild("HumanoidRootPart") or Player.Character.Humanoid.Health <= 0 then continue end
                    
                    local HRP = Player.Character.HumanoidRootPart
                    local found = false
                    
                    for _, obj in ipairs(workspace:GetDescendants()) do
                        if found then break end
                        if obj:IsA("Model") and obj:FindFirstChild("Humanoid") and obj.Humanoid.Health > 0 and not game.Players:GetPlayerFromCharacter(obj) then
                            local root = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Torso") or obj:FindFirstChild("Head")
                            if root then
                                local dist = (root.Position - HRP.Position).Magnitude
                                if dist <= getgenv().FarmRange then
                                    found = true
                                    HRP.CFrame = root.CFrame * CFrame.new(0, 5, -5)  -- Tele sát
                                    task.wait(0.1)
                                    
                                    -- Spam tool activate (cầm melee trước, game sẽ hit thật nếu sát)
                                    local tool = Player.Character:FindFirstChildOfClass("Tool")
                                    if tool then
                                        for i = 1, 10 do  -- Spam activate mạnh
                                            tool:Activate()
                                            task.wait(0.05)
                                        end
                                    else
                                        Rayfield:Notify({Title = "Cảnh báo", Content = "Cầm melee tool để kill thật! Không tool thì damage không work.", Duration = 8})
                                    end
                                    
                                    -- Backup: Sim attack key (nếu game dùng left click hoặc key)
                                    local vim = game:GetService("VirtualInputManager")
                                    vim:SendMouseButtonEvent(0, 0, 0, true, game, 0)  -- Left click down
                                    task.wait(0.1)
                                    vim:SendMouseButtonEvent(0, 0, 0, false, game, 0)  -- Up
                                    
                                    -- Check HP giảm (debug)
                                    local oldHP = obj.Humanoid.Health
                                    task.wait(0.5)
                                    if obj.Humanoid.Health < oldHP then
                                        print("Kill thành công: HP giảm từ " .. oldHP .. " xuống " .. obj.Humanoid.Health)
                                    else
                                        print("HP không giảm - Có thể anti-cheat block! Thử tool khác hoặc script raw.")
                                    end
                                end
                            end
                        end
                    end
                    
                    if not found then
                        Rayfield:Notify({Title = "Debug", Content = "Không thấy mob trong range " .. getgenv().FarmRange .. ". Bay gần khu spawn (Earth forest/quest) hoặc tăng range!", Duration = 6})
                    end
                end
            end)
        end
    end
})
